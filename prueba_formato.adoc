== 3. Bloque de Código Fuente

[source,python]

def calcular_potencia(base, exponente):
"""Calcula la potencia de un número usando recursividad."""
if exponente == 0:
return 1
else:
return base * calcular_potencia(base, exponente - 1)

print(f"Resultado: {calcular_potencia(2, 8)}")
